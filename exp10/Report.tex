\begin{abstract}
This report presents two single-image self-supervised denoising methods: Self2Self and Deep Image Prior (DIP). 
Self2Self leverages Bernoulli sampling and Dropout to train a denoising network using only noisy images, 
while DIP exploits the structure of a decoder-only network as an implicit prior for natural images. 
Both methods are evaluated on the Set12 dataset with noise levels $\sigma = 15, 25, 35, 50$, 
and compared with BM3D and Neighbor2Neighbor. 
Experimental results show that Self2Self achieves competitive performance among self-supervised methods, 
while DIP demonstrates the effectiveness of network architecture as a prior.
\end{abstract}

\begin{IEEEkeywords}
Image denoising, self-supervised learning, Self2Self, Deep Image Prior, single-image training
\end{IEEEkeywords}

% ============================================================
\section{Introduction}
% ============================================================
Image denoising is a fundamental low-level vision task. 
Traditional methods like BM3D rely on hand-crafted priors, while supervised deep learning methods require large paired datasets. 
Self-supervised methods such as Noise2Noise, Noise2Void, Neighbor2Neighbor, Self2Self, and DIP 
enable training with only noisy images, reducing the dependency on clean ground truth.

This report focuses on two single-image self-supervised methods:
\begin{itemize}
    \item \textbf{Self2Self}: Uses Bernoulli sampling to mask input pixels and Dropout for ensemble inference.
    \item \textbf{Deep Image Prior (DIP)}: Uses a fixed random input and a decoder network, with early stopping to prevent overfitting.
\end{itemize}

% ============================================================
\section{Related Work}
% ============================================================

\subsection{Noise2Noise and Neighbor2Neighbor}
Noise2Noise (N2N) trains on pairs of independently corrupted noisy images. 
Neighbor2Neighbor extends this idea by generating sub-image pairs through neighbor downsampling, 
enabling self-supervised training from single noisy images.

\subsection{Noise2Void and Self2Self}
Noise2Void (N2V) uses a blind-spot network where each pixel's prediction does not depend on its own value. 
Self2Self improves upon this by combining Bernoulli sampling with Monte Carlo Dropout, 
achieving better performance through ensemble inference.

\subsection{Deep Image Prior}
DIP demonstrates that the architecture of a convolutional network itself captures significant prior knowledge about natural images. 
By optimizing a randomly initialized decoder to fit a single noisy image, 
the network first learns the signal structure before overfitting to noise, 
making early stopping a natural regularization.

\section{Method}
% ============================================================

\subsection{Self2Self}

\subsubsection{Bernoulli Sampling}
Given a noisy image $y = x + n$, Self2Self generates a binary mask $M \in \{0, 1\}^{H \times W}$ 
where each element is independently sampled from a Bernoulli distribution with probability $p$:
\begin{equation}
    M_{i,j} \sim \text{Bernoulli}(p)
\end{equation}
The masked input is $\tilde{y} = y \odot (1 - M)$, where masked pixels are set to zero.

\subsubsection{Training Objective}
The network $f_\theta$ with Dropout is trained to minimize:
\begin{equation}
    \mathcal{L}(\theta) = \frac{1}{|M|} \sum_{i,j} M_{i,j} \left(f_\theta(\tilde{y})_{i,j} - y_{i,j}\right)^2
\end{equation}
The loss is computed only on masked positions, forcing the network to predict masked pixels from their neighbors.

\subsubsection{Ensemble Inference}
At test time, Monte Carlo Dropout is used for ensemble denoising:
\begin{equation}
    \hat{x} = \frac{1}{K} \sum_{k=1}^{K} f_{\theta}^{(k)}(y)
\end{equation}
where $K$ is the number of forward passes (typically $K=20$).

\subsubsection{Network Architecture}
The network is a U-Net with Dropout layers (dropout rate = 0.5) inserted after each ReLU activation.
Channel progression: 64 $\rightarrow$ 128 $\rightarrow$ 256 $\rightarrow$ 512 $\rightarrow$ 1024 $\rightarrow$ 512 $\rightarrow$ 256 $\rightarrow$ 128 $\rightarrow$ 64.

\subsection{Deep Image Prior (DIP)}

\subsubsection{Fixed Random Input}
DIP uses a fixed random tensor $z \sim \mathcal{N}(0, I)$ of small spatial size (e.g., $8 \times 8 \times 32$) as input.

\subsubsection{Decoder Architecture}
The network is a decoder-only architecture with upsampling layers:
\begin{equation}
    \hat{y} = f_\theta(z)
\end{equation}
The network progressively upsamples the random input to the target resolution.

\subsubsection{Optimization and Early Stopping}
The network is optimized to reconstruct the noisy image:
\begin{equation}
    \theta^* = \arg\min_\theta \|f_\theta(z) - y\|^2
\end{equation}
Due to the spectral bias of neural networks, the network first fits the low-frequency signal component 
before overfitting to high-frequency noise. Early stopping at the optimal iteration yields the denoised result.

\section{Analysis and Discussion}
% ============================================================

\subsection{Comparison of Self2Self and DIP}

\subsubsection{Similarities}
\begin{itemize}
    \item Both are \textbf{self-supervised} methods that require only noisy images for training.
    \item Both leverage \textbf{network architecture} as an implicit prior.
    \item Neither method requires paired clean-noisy training data.
\end{itemize}

\subsubsection{Differences}
\begin{itemize}
    \item \textbf{Training paradigm}: Self2Self trains on a dataset of noisy images; DIP optimizes per single image.
    \item \textbf{Inference speed}: Self2Self uses fast feed-forward inference (with ensemble); DIP requires per-image optimization at test time.
    \item \textbf{Regularization}: Self2Self uses Dropout + Bernoulli sampling; DIP relies on early stopping.
    \item \textbf{Scalability}: Self2Self generalizes to unseen images; DIP must re-optimize for each new image.
\end{itemize}

\subsection{Performance Analysis}
\begin{itemize}
    \item BM3D remains the strongest baseline, especially at low noise levels.
    \item Self2Self achieves competitive performance among deep learning self-supervised methods.
    \item DIP shows promising results but is computationally expensive due to per-image optimization.
    \item The gap between self-supervised and traditional methods is narrowing with improved architectures.
\end{itemize}

\subsection{Applicable Scenarios}
\begin{itemize}
    \item \textbf{Self2Self}: Suitable when a dataset of noisy images is available and fast inference is needed.
    \item \textbf{DIP}: Suitable for single-image scenarios where no training data is available, and computational cost is not a concern.
\end{itemize}

\section{Conclusion}
% ============================================================
This report implemented and compared Self2Self and DIP for single-image self-supervised denoising. 
Self2Self achieves better performance through ensemble inference with Monte Carlo Dropout, 
while DIP demonstrates the power of network architecture as an implicit prior. 
Both methods represent important directions in self-supervised image restoration.
